<body class="bg-slate-50 p-6">
    <div class="max-w-6xl mx-auto">
        <div id="statusHeader" class="bg-red-100 text-red-700 p-4 mb-4 rounded hidden">שגיאה בטעינת נתונים</div>
        <div class="bg-cyan-900 text-white p-8 rounded-2xl mb-8 flex justify-between items-center">
            <h1 class="text-2xl font-bold">דשבורד מועצה אזורית גוש עציון</h1>
            <select id="settlementFilter" onchange="updateUI()" class="bg-slate-800 p-2 rounded text-white border-none"></select>
        </div>

        <div class="grid grid-cols-1 md:grid-cols-3 gap-6 mb-8">
            <div class="bg-white p-6 rounded-xl shadow">
                <div class="text-sm text-gray-500">סה"כ תושבים</div>
                <div id="kpi-1" class="text-3xl font-bold italic">--</div>
            </div>
            <div class="bg-white p-6 rounded-xl shadow border-r-4 border-cyan-500">
                <div class="text-sm text-gray-500">פניות לחוסן</div>
                <div id="kpi-2" class="text-3xl font-bold italic">--</div>
            </div>
            <div class="bg-white p-6 rounded-xl shadow">
                <div class="text-sm text-gray-500">סטטוס: בטיפול</div>
                <div id="kpi-3" class="text-3xl font-bold italic">--</div>
            </div>
        </div>

        <div class="grid md:grid-cols-2 gap-8 h-96">
            <div class="bg-white p-6 rounded-xl shadow"><canvas id="chart1"></canvas></div>
            <div class="bg-white p-6 rounded-xl shadow"><canvas id="chart2"></canvas></div>
        </div>
    </div>

    <script>
        let dataHosen = [];
        let dataPop = {};
        let myCharts = {};

        async function loadData() {
            try {
                // טעינת חוסן
                const resH = await fetch('choshen.csv');
                const textH = await resH.text();
                dataHosen = Papa.parse(textH, {header:true, skipEmptyLines:true}).data;

                // טעינת אוכלוסייה
                const resP = await fetch('pop.csv');
                const textP = await resP.text();
                const rowsP = Papa.parse(textP).data;
                rowsP.forEach(r => { if(r[0]) dataPop[r[0].trim()] = r[1]; });

                initFilter();
                updateUI();
            } catch (e) {
                document.getElementById('statusHeader').classList.remove('hidden');
                document.getElementById('statusHeader').innerText = "שגיאה: וודא שהקבצים choshen.csv ו-pop.csv נמצאים בתיקייה";
            }
        }

        function initFilter() {
            const select = document.getElementById('settlementFilter');
            const cities = [...new Set(dataHosen.map(d => d.עיר).filter(Boolean))].sort();
            select.innerHTML = '<option value="all">כל הגוש</option>' + 
                cities.map(c => `<option value="${c}">${c}</option>`).join('');
        }

        function updateUI() {
            const city = document.getElementById('settlementFilter').value;
            const filtered = city === 'all' ? dataHosen : dataHosen.filter(d => d.עיר === city);

            document.getElementById('kpi-1').innerText = dataPop["מספר התושבים הרשומים במערכת"] || "53,134";
            document.getElementById('kpi-2').innerText = filtered.length;
            document.getElementById('kpi-3').innerText = filtered.filter(d => d['סטטוס מטופל/ת'] === 'בטיפול').length;

            render(filtered);
        }

        function render(data) {
            const stats = {};
            data.forEach(d => { const s = d['סטטוס מטופל/ת'] || 'אחר'; stats[s] = (stats[s] || 0) + 1; });

            if(myCharts.c1) myCharts.c1.destroy();
            myCharts.c1 = new Chart(document.getElementById('chart1'), {
                type: 'pie',
                data: { labels: Object.keys(stats), datasets: [{ data: Object.values(stats), backgroundColor: ['#0891b2', '#10b981', '#f59e0b', '#ef4444'] }] },
                options: { maintainAspectRatio: false, plugins: { legend: { position: 'bottom', rtl: true } } }
            });
        }

        window.onload = loadData;
    </script>
